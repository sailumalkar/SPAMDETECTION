import java.util.*;

public class SpamDetector {
    // List of spam keywords
    private static final Set<String> SPAM_KEYWORDS = new HashSet<>(Arrays.asList(
            "win", "free", "prize", "lottery", "click here", "subscribe", "claim", "money", "urgent", "offer"
    ));

    // Function to check if a message is spam
    public static boolean isSpam(String message) {
        String lowerMessage = message.toLowerCase();
        for (String keyword : SPAM_KEYWORDS) {
            if (lowerMessage.contains(keyword)) {
                return true; // Spam detected
            }
        }
        return false; // Not spam
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.println("Enter a message:");
        String message = scanner.nextLine();
        
        if (isSpam(message)) {
            System.out.println("⚠️ Spam detected!");
        } else {
            System.out.println("✅ Message is safe.");
        }

        scanner.close();
    }
}
